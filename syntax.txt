stm = vardec | assignment | funcall | propertyaccess | propertycall | arrayacess | arrayassignment | value
vardec = var [id] = type
assignment = [id] = type
funcall = [id]: [stm] [args] [args]...
propertyaccess = [id].[property]
propertycall = [id].[property]: .*
arrayaccess = id[[index]]
arrayassignment = id[[index]] = value
value = int | string | arr | func | object
int = \d+
string = ".*"
arr = [arg,arg,arg,arg] | a..b | \d+..\d+ | null
